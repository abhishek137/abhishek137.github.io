/* Solution to HackerRank Problem: Birthday Chocolate
 * https://www.hackerrank.com/challenges/the-birthday-bar
 */

import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    static int solve(int n, int[] s, int d, int m){
        // Complete this function

        int count = 0;

        for(int i = 0; i < s.length; i++){

            if((i + m - 1) == s.length){
                break;
            }
            else{
                int sum = 0;
                for(int j = 0; j < m ; j ++){
                    sum += s[i+j];
                }

                if(sum == d){
                    count++;
                }
            }
        }

        return count;
    }

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int[] s = new int[n];
        for(int s_i=0; s_i < n; s_i++){
            s[s_i] = in.nextInt();
        }
        int d = in.nextInt();
        int m = in.nextInt();
        int result = solve(n, s, d, m);
        System.out.println(result);
    }
}

Solution to HackerRank Problem - Breaking the Records

2017-07-29
hackerrank, java, solution 
/* Solution to HackerRank Problem: Breaking the Records
 * https://www.hackerrank.com/challenges/breaking-best-and-worst-records
 */

import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    static int[] getRecord(int[] s){
        // Complete this function
        int[] result = new int[2];

        int max  = s[0];
        int min = s[0];

        for (int i=1; i < s.length ; i++){
            if(s[i]>max){
                result[0]++;
                max = s[i];
            }

            if(s[i]<min){
                result[1]++;
                min = s[i];
            }
        }

        return result;
    }

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int[] s = new int[n];
        for(int s_i=0; s_i < n; s_i++){
            s[s_i] = in.nextInt();
        }
        int[] result = getRecord(s);
        String separator = "", delimiter = " ";
        for (Integer val : result) {
            System.out.print(separator + val);
            separator = delimiter;
        }
        System.out.println("");
    }
}

/* Solution to HackerRank Problem: Minimum Absolute Difference in an Array
 * https://www.hackerrank.com/challenges/minimum-absolute-difference-in-an-array
 */

import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int[] a = new int[n];
        for(int a_i=0; a_i < n; a_i++){
            a[a_i] = in.nextInt();
        }

        int min = Integer.MAX_VALUE;

        Arrays.sort(a);

        for (int i = 0; i < a.length-1; i++){            
            min = Math.min(Math.abs(a[i]- a[i+1]), min);
        }

        System.out.println(min);
    }
}

Solution to HackerRank Problem - Grading Students

/* Solution to HackerRank Problem: Grading Students
 * https://www.hackerrank.com/challenges/grading
 */

import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    static int[] solve(int[] grades){
        for(int i = 0; i < grades.length; i++){
            grades[i] = grade(grades[i]);
        }

        return grades;
    }

    static int grade(int m){
        if(m < 38){
            return m;
        }
        else{
            int nm = nextMultiple(m);

            if((nm - m) < 3){
                return nm;
            }
            else{
                return m;
            }

        }
    }

    static int nextMultiple(int n){
        int m = n/5;

        return (m+1)*5;
    }

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int[] grades = new int[n];
        for(int grades_i=0; grades_i < n; grades_i++){
            grades[grades_i] = in.nextInt();
        }
        int[] result = solve(grades);
        for (int i = 0; i < result.length; i++) {
            System.out.print(result[i] + (i != result.length - 1 ? "\n" : ""));
        }
        System.out.println("");


    }
}

Solution to HackerRank Problem - Algorithmic Crush

2017-04-30
hackerranksolution
hackerrank, java, solution

Solution: Greedy Approach

Test cases #7-13 timed out with this solution.

import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {

        Scanner in = new Scanner(System.in);

        int N = in.nextInt();
        int M = in.nextInt();

        long[] arr = new long[N];

        for(int m=0; m < M; m++){
            int a = in.nextInt()-1;
            int b = in.nextInt()-1;
            int k = in.nextInt();

            // update each element between a and b
            while(a<=b){
                arr[a] += k;                 
                a++;
            }
        }

        long max = 0;

        // find the element with maximum value
        for(int i=0; i < N; i++){
            if(arr[i] > max){
                max = arr[i];
            }
        }

        System.out.println(max);
    }
}

Solution: Relative Difference

Instead of storing the actual value of each element in the array, this solution stores the difference between current element and previous element.

/* Solution to HackerRank Problem: Algorithmic Crush
 * URL: https://www.hackerrank.com/challenges/crush
 */

import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {

        Scanner in = new Scanner(System.in);

        int N = in.nextInt();
        int M = in.nextInt();

        // need an extra elemtent in array to deal with updating element at index b
        long[] arr = new long[N+1];

        for(int m=0; m < M; m++){
            int a = in.nextInt()-1;
            int b = in.nextInt()-1;
            int k = in.nextInt();

            // update element at index a. Difference between further element till index b will be same as k is added to all these elements
            arr[a] += k;

            // update element at index b+1 as difference between element at index b and b+1 has changed after adding k to element at index b
            arr[b+1] -= k;
        }

        long max = 0;
        long sum = 0;

        // find the element with maximum value by summing up the difference
        for(int i=0; i < N; i++){
            sum += arr[i];

            if(sum > max){
                max = sum;
            }
        }

        System.out.println(max);
    }
}

Solution to HackerRank Problem - Nikita and the Game

/* Solution to HackerRank Problem: Nikita and the Game
 * URL: https://www.hackerrank.com/challenges/array-splitting
 */

import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);

        int T = in.nextInt();

        for(int t=0; t < T; t++){
            int N = in.nextInt();

            int[] a  = new int[N];

            for(int n=0; n < N; n++){
                a[n] = in.nextInt();
            }

            System.out.println(split(a));
        }
    }

    public static int split(int[] o){
        int length = o.length;

        if(length == 1){
            return 0;
        }

        // array with odd sum cannot be splitted into two parts of equal sum
        int total = sum(o);        
        if(total % 2 == 1){
            return 0;
        }

        int sum = o[0];
        for(int index=1; index < length; index++){

            // iterate till sum of first part is twice the total sum i.e both parts have equal sum
            if(sum * 2 == total){
                int[] o1 = Arrays.copyOfRange(o, 0, index);
                int[] o2 = Arrays.copyOfRange(o, index, length);                

                // consider the part which can be split for maximum times
                return Math.max(split(o1),split(o2))+1;
            }    

            sum += o[index];

        }

        return 0;
    }

    public static int sum(int[] a){
        int sum = 0;

        for(int i : a){
            sum += i;
        }

        return sum;
    }
}
