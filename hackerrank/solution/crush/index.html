Solution: Greedy Approach

Test cases #7-13 timed out with this solution.

import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {

        Scanner in = new Scanner(System.in);

        int N = in.nextInt();
        int M = in.nextInt();

        long[] arr = new long[N];

        for(int m=0; m < M; m++){
            int a = in.nextInt()-1;
            int b = in.nextInt()-1;
            int k = in.nextInt();

            // update each element between a and b
            while(a<=b){
                arr[a] += k;                 
                a++;
            }
        }

        long max = 0;

        // find the element with maximum value
        for(int i=0; i < N; i++){
            if(arr[i] > max){
                max = arr[i];
            }
        }

        System.out.println(max);
    }
}

Solution: Relative Difference

Instead of storing the actual value of each element in the array, this solution stores the different between current element and previous element.

/* Solution to HackerRank Problem: Algorithmic Crush
 * URL: https://www.hackerrank.com/challenges/crush
 */

import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {

        Scanner in = new Scanner(System.in);

        int N = in.nextInt();
        int M = in.nextInt();

        // need an extra elemtent in array to deal with updating element at index b
        long[] arr = new long[N+1];

        for(int m=0; m < M; m++){
            int a = in.nextInt()-1;
            int b = in.nextInt()-1;
            int k = in.nextInt();

            // update element at index a. Difference between further element till index b will be same as k is added to all these elements
            arr[a] += k;

            // update element at index b+1 as difference between element at index b and b+1 has changed after adding k to element at index b
            arr[b+1] -= k;
        }

        long max = 0;
        long sum = 0;

        // find the element with maximum value by summing up the difference
        for(int i=0; i < N; i++){
            sum += arr[i];

            if(sum > max){
                max = sum;
            }
        }

        System.out.println(max);
    }
}
