Javascript Closures

2017-05-03
topic
interview, javascript, nodejs

What is a closure?

  • A closure is a function defined within another function that has access to all the variables within the outer function scope even after the outer function returns.
  • It makes it possible for a function to have “private” variables that can maintain their state. (The idea here is similar to instance variables in java)

A classic implementation


function greet(name){
    // private variable
    var message = "Hello " + name + "!!!";

    // a closure : function defined within another function
    function getMessage(){
        // has access to variables withint the outer function scope
        return message;
    }

    // return closure so that it can be accesible outside of this scope;
    return getMessage();
}

console.log(greet('Daredevil')); //Hello Daredevil!!!

Let’s discuss the concept of closure by implementing a counter that can be used from any part of the application.

Approach: Variable outside object

var count = 0;

// functions inside an javascript object
var counter = {
    add: function(){
        count += 1;
    },
    getCount: function(){
        return count;
    },
    reset: function(){
        count = 0;
    }
}

console.log(typeof counter); // object

counter.add();
counter.add();

console.log(counter.getCount()); // 2

count = 0;

console.log(counter.getCount()); // 0

The counter can be reset without calling reset function.

Approach: Variable inside object

// functions inside an javascript object
var counter = {
    count: 0,
    add: function(){
        this.count += 1;
    },
    getCount: function(){
        return this.count;
    },
    reset: function(){
        this.count = 0;
    }
}

console.log(typeof counter); // object

counter.add();
counter.add();

console.log(counter.getCount()); // 2

counter.count = 0;

console.log(counter.getCount()); // 0

The counter can be reset without calling reset function.

Approach: Variable inside a function

// functions inside an javascript object
var counter = new function(){
    // private variable
    var count = 0;

    // a closure : function with another function with access to private variable
    function add(){
        count += 1;
    }

    // another closure
    function getCount(){
        return count;
    }

    // another closure
    function reset(){
        count = 0;
    }

    // all the closures are private by default, make them accessible outside of this scope by returning as a object
    return {
        add: add,
        getCount: getCount,
        reset: reset
    }
};

console.log(typeof counter); // object

counter.add();
counter.add();

console.log(counter.getCount()); // 2

counter.count = 0;

console.log(counter.getCount()); // 2

counter.reset();

console.log(counter.getCount()); // 0

The counter can be reset only by calling reset function. Variable count is truly private and cannot be accessed outside of the function scope.

Bonus: Multiple Counter instances

It is possible to acheive “object oritented” implementation using closures. Javascript design patterns like Module Pattern and Revealing Module Pattern are based on closures.

// functions inside an javascript object
var Counter = function(){
    // private variable
    var count = 0;

    // another way of writing closures that are accessible outside of this scope
    return {
        add: function (){
            count += 1;
        },
        getCount: function(){
            return count;
        },
        reset: function(){
            count = 0;
        }
    }
};

console.log(typeof Counter); // function

var counter1 = new Counter();
var counter2 = new Counter();

console.log(typeof counter1); // object

counter1.add();
counter1.add();
counter2.add();

console.log(counter1.getCount()); // 2
console.log(counter2.getCount()); // 1

counter1.reset();

console.log(counter1.getCount()); // 0
console.log(counter2.getCount()); // 1
